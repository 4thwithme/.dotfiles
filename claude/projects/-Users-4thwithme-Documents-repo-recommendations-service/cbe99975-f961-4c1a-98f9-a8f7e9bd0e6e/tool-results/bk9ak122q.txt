+import "server-only";
+
+import { cacheLife, cacheTag } from "next/cache";
+import { config } from "@/config";
+import type {
+  RelatedProductsResponse,
+  RecommendationsResponse,
+  RecoBrandsResponse,
+} from "./client";
+
+export async function getRelatedStyles(
+  styleId: string,
+  colorId: string,
+): Promise<RelatedProductsResponse | null> {
+  "use cache: remote";
+  cacheLife("hours");
+  cacheTag(`related-styles-${styleId}-${colorId}`);
+
+  try {
+    const params = new URLSearchParams({ PK: colorId });
+    const res = await fetch(
+      `${config.server.CATALOG_BASE_URL}/products/styles/${styleId}/related_styles.json?${params}`,
+    );
+    if (!res.ok) return null;
+    return await res.json();
+  } catch {
+    return null;
+  }
+}
+
+export async function getRecommendedStyles(
+  styleId: string,
+  colorId: string,
+  limit: number = 30,
+): Promise<RecommendationsResponse | null> {
+  "use cache: remote";
+  cacheLife("hours");
+  cacheTag(`recommended-styles-${styleId}`);
+
+  try {
+    const params = new URLSearchParams({
+      colorId,
+      limit: String(limit),
+    });
+    const res = await fetch(
+      `${config.server.RECO_API_URL}/api/v1/related-products/${styleId}?${params}`,
+    );
+    if (!res.ok) return null;
+    return await res.json();
+  } catch {
+    return null;
+  }
+}
+
+export async function getRecoBrands(
+  styleId: string,
+  colorId: string,
+  brand: string,
+  limit: number = 30,
+): Promise<RecoBrandsResponse | null> {
+  "use cache: remote";
+  cacheLife("hours");
+  cacheTag(`reco-brands-${styleId}-${brand}`);
+
+  try {
+    const params = new URLSearchParams({
+      brand,
+      colorId,
+      limit: String(limit),
+      styleId,
+      sortBy: "rating",
+    });
+    const res = await fetch(
+      `${config.server.RECO_API_URL}/api/v1/related-products/brand?${params}`,
+    );
+    if (!res.ok) return null;
+    return await res.json();
+  } catch {
+    return null;
+  }
