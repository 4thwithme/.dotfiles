+/**
+ * Lightweight fetch mock decorator — no MSW required.
+ *
+ * In any story, set `parameters.fetchMock` to a map of URL substring → response body:
+ *
+ * ```ts
+ * parameters: {
+ *   fetchMock: {
+ *     '/products/quotes/new': MOCK_QUOTES_API_RESPONSE,
+ *   },
+ * },
+ * ```
+ *
+ * Requests whose URL contains the key are intercepted and return the mock JSON.
+ * All other requests fall through to the real `window.fetch`.
+ * The original fetch is restored when the story unmounts.
+ */
+const FetchMockWrapper = ({ Story, handlers }: { Story: React.ComponentType; handlers: Record<string, unknown> }) => {
+  // Patch synchronously during render so the mock is in place before child effects fire.
+  // (React runs child useEffects before parent useEffects, so patching in useEffect is too late.)
+  const originalFetchRef = React.useRef(window.fetch);
+  window.fetch = async (input: RequestInfo | URL, init?: RequestInit) => {
+    const url = input.toString();
+    for (const [pattern, responseBody] of Object.entries(handlers)) {
+      if (url.includes(pattern)) {
+        return new Response(JSON.stringify(responseBody), {
+          status: 200,
+          headers: { 'Content-Type': 'application/json' },
+        });
+      }
+    }
+    return originalFetchRef.current(input, init);
+  };
+
+  useEffect(() => {
+    return () => {
+      window.fetch = originalFetchRef.current;
+    };
+  }, []);
+
+  return <Story />;
+};
+
+export const withFetchMock: Decorator = (Story, { parameters }) => {
+  const handlers = parameters.fetchMock as Record<string, unknown> | undefined;
+  if (!handlers) return <Story />;
+  return <FetchMockWrapper Story={Story} handlers={handlers} />;
+};

+# RECO-730: TrendingProductsWidget Migration Plan
+
+**Branch:** `RECO-730-trendng-products-migration`
+**Package:** `@customink/pigment-extensions/reco`
+**Source:** `catalog/app/javascript/features/TrendingProducts/TrendingCarousel.jsx`
+**References:** [ADR-20](../adr/20-Extensions-Package.md) · [RFC-1](../rfc/1-trending-products-widget-migration.md) · PR #1068 · PR #1083

+---
+
+## Scope Summary
+
+Migrate `TrendingCarousel.jsx` from the catalog repo into `packages/extensions/reco/TrendingProductsWidget/` using the ADR-20 extensions structure. The widget is broken into internal atoms/molecules (not publicly exported) and one assembled widget component + one dumb hook (publicly exported).
+
+---
+
+## What Does NOT Move
+
+| Catalog piece | Why | How replaced |
+|---|---|---|
+| `FavoritesProvider` / `useFavorites` | Auth-aware, catalog-specific context | Catalog wraps widget with `FavoritesProvider`; injects button via `renderFavoriteButton` render prop |
+| `ProductCardPriceDetails` | Deeply catalog-specific: Rollbar, AB tests, `useProductCardPriceDetails` data fetching | `RecommendationsProductCard` exposes `renderPriceDetails` render prop |
+| `useTrendingProductsWrapper` / `usePromotionalProducts` | React Query, catalog API config | Replaced by dumb `useTrendingProducts` hook that accepts a `fetcher` function |
+| `firePromoTrendingRecommendationClickEvent` / `ViewEvent` | Catalog analytics utils | Replaced by `onWidgetViewed` / `onCardViewed` / `onCardClicked` callback props on the widget |
+| `useBreakpoint` (catalog hook) | Catalog-specific | Re-implement as local `useMediaQuery`-based hook inside widget |
+| MUI `styled()` | v0 pattern, no MUI in new code | CSS Modules |
+

+## useTrendingProducts Hook (public)
+
+```ts
+type TrendingProductsFetcher = () => Promise<TrendingProduct[]>;
+
+type UseTrendingProductsOptions = {
+  fetcher: TrendingProductsFetcher;
+  enabled?: boolean;
+};
+
+type UseTrendingProductsResult = {
+  products: TrendingProduct[];
+  isLoading: boolean;
+  isError: boolean;
+};
+
+function useTrendingProducts(options: UseTrendingProductsOptions): UseTrendingProductsResult;
+```
+
+**Implementation notes:**
+- Dumb hook: calls `fetcher()`, manages `loading`/`error`/`data` state with `useEffect` + `useState`
+- No React Query, no catalog API wiring — those stay in catalog
+- No in-memory cache for phase 1 (simple `useEffect`, re-fetches on mount)
+- `enabled` flag: skip fetch when `false` (matches catalog `useTrendingProducts` pattern)
+- Returns empty `products: []` on error (same behavior as current catalog hook)
+
+**Catalog usage:**
+```ts
+// catalog: useTrendingProductsWrapper.js wraps this
+const { products, isLoading } = useTrendingProducts({
+  fetcher: () => fetchPromotionalProducts({ categoryId, limit: 20 }),
+  enabled: true,
+});
+```

+## TrendingProductsWidget Assembly
+
+**Responsive slidesPerView logic** (preserve from `TrendingCarousel.jsx`):
+
+| Breakpoint | Products count | slidesPerView | spacing |
+|---|---|---|---|
+| Mobile (`≤ 768px`) | ≥ 3 | 1.6 | 8px |
+| Mobile | < 3 | `'auto'` | 8px |
+| Desktop | ≥ 5 | 4 | 16px |
+| Desktop | < 5 | `'auto'` | 16px |
+
+**Arrow visibility:** `showArrows = slidesPerView !== 'auto' && !isMobile`
+
+**Widget behavior:**
+- `isLoading` → render `LoadingSkeleton`
+- `products.length === 0` → return `null`
+- Viewport visibility: `IntersectionObserver` on wrapper → `onWidgetViewed(products)` fires once
+- Card click → `onCardClicked(product, index)`
+- Card in viewport → `onCardViewed(product, index)` (per-card IntersectionObserver, optional phase 2)
+
+**`FavoritesProvider` is NOT wrapped here** — catalog is responsible for wrapping:
+```tsx
+// catalog (after migration)
+<FavoritesProvider pageType={PageType.SUBCATEGORY}>
+  <TrendingProductsWidget
+    products={products}
+    isLoading={isLoading}
+    renderFavoriteButton={(product) => (
+      <FavoriteButton
+        styleId={product.styleId}
+        isFavorited={isFavorited(product.styleId)}
+        onToggle={...}
+      />
+    )}
+    renderPriceDetails={(product) => (
+      <ProductCardPriceDetails
+        defaultUnitPrices={product.defaultUnitPrices}
+        productId={product.styleId}
+        quantity={product.defaultQuoteQty}
+        {...catalogSpecificProps}
+      />
+    )}
+  />
+</FavoritesProvider>
+```

+import React, { useCallback, useEffect, useRef, useState } from 'react';
+import { createPortal } from 'react-dom';
+import styles from './Popover.module.css';
+
+type Side = 'top' | 'bottom' | 'left' | 'right';
+
+const GAP_PX = 8;
+const VP_PAD = 8; // minimum gap between panel edge and viewport edge
+
+// Off-screen sentinel: keeps the panel in the DOM for fade-out but outside the viewport.
+const OFFSCREEN: React.CSSProperties = { position: 'fixed', top: -9999, left: -9999 };
+
+function clamp(value: number, min: number, max: number): number {
+  return Math.max(min, Math.min(max, value));
+}
+
+/** Returns a viewport-clamped `position:fixed` style. Reads offsetWidth/offsetHeight while the panel is off-screen for single-pass positioning. */
+function computeClampedStyle(triggerRect: DOMRect, side: Side, panelW: number, panelH: number): React.CSSProperties {
+  const vw = window.innerWidth;
+  const vh = window.innerHeight;
+
+  switch (side) {
+    case 'bottom': {
+      const left = clamp(triggerRect.left + triggerRect.width / 2, VP_PAD + panelW / 2, vw - VP_PAD - panelW / 2);
+      const top = clamp(triggerRect.bottom + GAP_PX, VP_PAD, vh - VP_PAD - panelH);
+      return { position: 'fixed', top, left };
+    }
+    case 'top': {
+      const left = clamp(triggerRect.left + triggerRect.width / 2, VP_PAD + panelW / 2, vw - VP_PAD - panelW / 2);
+      const top = clamp(triggerRect.top - GAP_PX - panelH, VP_PAD, vh - VP_PAD - panelH);
+      return { position: 'fixed', top, left };
+    }
+    case 'right': {
+      const left = clamp(triggerRect.right + GAP_PX, VP_PAD, vw - VP_PAD - panelW);
+      const top = clamp(triggerRect.top + triggerRect.height / 2, VP_PAD + panelH / 2, vh - VP_PAD - panelH / 2);
+      return { position: 'fixed', left, top };
+    }
+    case 'left': {
+      const left = clamp(triggerRect.left - GAP_PX - panelW, VP_PAD, vw - VP_PAD - panelW);
+      const top = clamp(triggerRect.top + triggerRect.height / 2, VP_PAD + panelH / 2, vh - VP_PAD - panelH / 2);
+      return { position: 'fixed', left, top };
+    }
+  }

+  /** Rich content rendered inside the popover panel */
+  content: React.ReactNode;
+  /** The element that triggers the popover on hover/focus */
+  children: React.ReactNode;
+  /**
+   * Placement relative to the trigger.

+   * @default 'auto'

+
+  useEffect(() => {
+    setMounted(true);
+  }, []);
+  const [resolvedPlacement, setResolvedPlacement] = useState<Side>('bottom');
+  const [panelStyle, setPanelStyle] = useState<React.CSSProperties>(OFFSCREEN);
+
+  const wrapperRef = useRef<HTMLSpanElement>(null);
+  const panelRef = useRef<HTMLSpanElement>(null);
+

+  const setWrapperRef = useCallback(

+      (wrapperRef as React.MutableRefObject<HTMLSpanElement | null>).current = node;
+      if (typeof ref === 'function') ref(node);
+      else if (ref) (ref as React.MutableRefObject<HTMLSpanElement | null>).current = node;
+    },
+    // ref identity is stable for the lifetime of the component
+    // eslint-disable-next-line react-hooks/exhaustive-deps
+    [],

+        bottom: window.innerHeight - triggerRect.bottom,
+        top: triggerRect.top,
+        right: window.innerWidth - triggerRect.right,
+        left: triggerRect.left,
+      };
+      side = (Object.entries(spaces) as [Side, number][]).reduce((a, b) => (b[1] > a[1] ? b : a))[0];
+    } else {
+      side = placement;
+    }
+    setResolvedPlacement(side);

+    const panelH = panelRef.current.offsetHeight;

+  }, [placement]);

+      className={styles.wrapper}

+      onMouseLeave={hide}
+      onFocus={show}
+      onBlur={hide}

+            </span>,
+            document.body,
+          )
+        : null}
+    </span>

+import { useEffect, useState } from 'react';
+import units from '@customink/design-tokens/semantics/units';
+import type { Breakpoint, UnitPrice } from './types';
+
+const BREAKPOINTS = {
+  mobile: parseInt(units.size.breakpoint.tablet, 10),
+  desktop: parseInt(units.size.breakpoint.desktop, 10),
+};
+
+function getBreakpoint(width: number): Breakpoint {
+  if (width < BREAKPOINTS.mobile) return 'mobile';
+  if (width < BREAKPOINTS.desktop) return 'tablet';
+  return 'desktop';
+}
+
+export const DEFAULT_SLIDES_CONFIG: Record<Breakpoint, { perView: number; spacing: number }> = {
+  mobile: { perView: 2.25, spacing: 8 },
+  tablet: { perView: 4, spacing: 16 },
+  desktop: { perView: 5, spacing: 16 },
+};
+
+export function getSlidesConfig(
+  breakpoint: Breakpoint,
+  overrides?: Partial<Record<Breakpoint, { perView: number; spacing: number }>>,
+): { perView: number; spacing: number } {
+  return { ...DEFAULT_SLIDES_CONFIG[breakpoint], ...overrides?.[breakpoint] };
+}
+
+export function useBreakpoint(): Breakpoint {
+  const [breakpoint, setBreakpoint] = useState<Breakpoint>(() =>
+    typeof window !== 'undefined' ? getBreakpoint(window.innerWidth) : 'desktop',
+  );
+
+  useEffect(() => {
+    const mqMobile = window.matchMedia(`(max-width: ${BREAKPOINTS.mobile - 1}px)`);
+    const mqTablet = window.matchMedia(`(min-width: ${BREAKPOINTS.mobile}px) and (max-width: ${BREAKPOINTS.desktop - 1}px)`);
+    const handler = () => setBreakpoint(getBreakpoint(window.innerWidth));
+    mqMobile.addEventListener('change', handler);
+    mqTablet.addEventListener('change', handler);
+    return () => {
+      mqMobile.removeEventListener('change', handler);
+      mqTablet.removeEventListener('change', handler);
+    };
+  }, []);
+
+  return breakpoint;
+}
+
+export function formatRatingScore(score: number): string {
+  const adjusted = score < 1 ? 1 : score;
+  return adjusted.toFixed(1);
+}
+
+export function formatRatingCount(count: number): string | null {
+  if (Number.isNaN(count)) return null;
+  const cap = 10_000;
+  const capped = count > cap ? cap : count;
+  const formatted = capped.toLocaleString();
+  return `(${count > cap ? `${formatted}+` : formatted})`;
+}
+
+export function formatPrice(
+  unitPrices: UnitPrice[],
+  quoteQty: number,
+): { amount: string; suffix: string; originalAmount?: string } | null {
+  if (!unitPrices.length) return null;
+  const match = unitPrices.find((p) => p.quantity === quoteQty) ?? unitPrices[0];
+  const isDiscounted = match.regularPrice != null && match.regularPrice > match.price;
+  return {
+    amount: `$${match.price.toFixed(2)}`,
+    suffix: `/ea for ${match.quantity} items`,
+    ...(isDiscounted && match.regularPrice != null && { originalAmount: `$${match.regularPrice.toFixed(2)}` }),
+  };
+}

+/** Default YDH logo design used for recommendation product images */
+export const DEFAULT_RECOMMENDATIONS_DESIGN_CID = 'djn0-00by-wu36';

+/**
+ * Applies logo-placement query params to a product image URL.
+ * @param designCid - Design CID to embed (e.g. `DEFAULT_RECOMMENDATIONS_DESIGN_CID` or a real design CID).
+ */
+export function processImageUrl(url: string | null | undefined, designCid: string): string | undefined {
+  if (!url) return undefined;
+  try {
+    const imageUrl = new URL(url);
+    imageUrl.searchParams.delete('design');
+    imageUrl.searchParams.set('design', designCid);
+    imageUrl.searchParams.set('placeMax', '1');
+    imageUrl.searchParams.set('placeMaxPct', '0.8');
+    imageUrl.searchParams.set('placeUseProduct', '1');
+    imageUrl.searchParams.set('placeUseView', 'front');
+    return imageUrl.toString();
+  } catch {
+    return url;
+  }

+import type { TrendingProduct, UnitPrice } from '../ProductsCarouselWidget/types';
+import type { ApiRecoProduct, ApiUnitPrice } from './types';
+import { processImageUrl } from './imageUtils';
+
+export type MapProductOptions = {
+  /** Design CID for logo placement in image URLs. When omitted, raw URL is used. */
+  imageDesignCid?: string;
+};
+
+export function mapUnitPrice(p: ApiUnitPrice): UnitPrice {
+  const price = parseFloat(p.price);
+  const regularPrice = p.regular_unit_price != null ? parseFloat(p.regular_unit_price) : null;
+  return {
+    quantity: p.quantity,
+    price,
+    ...(regularPrice != null && regularPrice > price && { regularPrice }),
+  };
+}
+
+export function mapProduct(p: ApiRecoProduct, { imageDesignCid }: MapProductOptions = {}): TrendingProduct {
+  const ratingScore = typeof p.rating_score === 'number' ? p.rating_score : parseFloat(p.rating_score);
+  const ratingCount = parseInt(p.rating_count, 10);
+
+  return {
+    styleId: p.style_id,
+    name: p.name,
+    link: p.link,
+    imageSrc: imageDesignCid ? processImageUrl(p.image_src, imageDesignCid) ?? p.image_src : p.image_src,
+    minQty: p.min_qty,
+    ratingScore: ratingScore > 0 ? ratingScore : undefined,
+    ratingCount: ratingCount > 0 ? ratingCount : undefined,
+    defaultUnitPrices: (p.default_unit_prices ?? []).map(mapUnitPrice),
+    defaultQuoteQty: p.default_quote_qty,
+    primaryCategoryId: p.primary_category_id ?? undefined,
+  };
+}

+/**
+ * Computes the quantity tiers shown in the price details table.
+ * Ported from catalog: utils/priceGridQuantities.js
+ */
+export function priceGridQuantities(bulkMinimumQty: number, isSinglesEnabled = false, isSpecialty = false): number[] {
+  const SINGLES_ENABLED_APPAREL = [1, 10, 25, 50, 100, 500];
+  const SPECIALTY_TINY_MINIMUMS = [2, 4, 6, 8, 10];
+  const DEFAULT_LESSER_MINIMUMS = [1, 2, 4, 8, 17, 83];
+  const DEFAULT_HIGHER_MINIMUMS = [1, 2, 4, 10, 20, 50];
+
+  if (isSinglesEnabled) return SINGLES_ENABLED_APPAREL;
+
+  const minQty = parseInt(String(bulkMinimumQty), 10);
+
+  if (minQty > 0) {
+    if (isSpecialty && minQty < 6) {
+      return [minQty, ...SPECIALTY_TINY_MINIMUMS.map((mult) => Math.ceil((mult * minQty) / 10) * 10)];
+    }
+    if (minQty < 25) {
+      return DEFAULT_LESSER_MINIMUMS.map((mult) => mult * minQty);
+    }
+    return DEFAULT_HIGHER_MINIMUMS.map((mult) => mult * minQty);
+  }
+
+  return [];
+}

+import { useEffect, useMemo, useState } from 'react';
+import { priceGridQuantities } from './priceGridQuantities';
+
+export type UseProductCardPriceDetailsOptions = {
+  styleId: number;
+  minQty: number;
+  /** Included in the quantity tiers alongside the standard grid. Typically defaultQuoteQty from TrendingProduct. */
+  defaultQuoteQty: number;
+  /** Number of front print colors. @default 1 */
+  frontColor?: number;
+  /** Number of back print colors. @default 0 */
+  backColor?: number;
+  /** Whether singles (qty 1) are enabled for this product. @default false */
+  isSinglesEnabled?: boolean;
+  /** Whether this is a specialty product with tiny minimums. @default false */
+  isSpecialty?: boolean;
+  /** Catalog quotes endpoint path. @default '/products/quotes/new' */
+  quotesPath?: string;
+  /**
+   * Postal code sent with the quote request for shipping-zone pricing.
+   * Defaults to a neutral US zip ('22031') for consistent catalog pricing.
+   */
+  postalCode?: string;
+};
+
+export type UseProductCardPriceDetailsResult = {
+  isLoading: boolean;
+  isError: boolean;
+  /** Map of quantity → formatted price string (e.g. "27.44"), null when price unavailable. */
+  priceDetailsData: Record<number, string | null> | null;
+  /** Sorted list of quantities shown in the table. */
+  quotableQuantities: number[];
+};
+
+export function useProductCardPriceDetails({
+  styleId,
+  minQty,
+  defaultQuoteQty,
+  frontColor = 1,
+  backColor = 0,
+  isSinglesEnabled = false,
+  isSpecialty = false,
+  quotesPath = '/products/quotes/new',
+  postalCode = '22031',
+}: UseProductCardPriceDetailsOptions): UseProductCardPriceDetailsResult {
+  const [priceDetailsData, setPriceDetailsData] = useState<Record<number, string | null> | null>(null);
+  const [isLoading, setIsLoading] = useState(true);
+  const [isError, setIsError] = useState(false);
+
+  const quotableQuantities = useMemo(() => {
+    const grid = priceGridQuantities(minQty, isSinglesEnabled, isSpecialty);
+    return [...new Set([defaultQuoteQty, ...grid])].sort((a, b) => a - b);
+  }, [minQty, defaultQuoteQty, isSinglesEnabled, isSpecialty]);
+
+  // Stable key so the effect only re-runs when quantities actually change
+  const quantitiesKey = quotableQuantities.join(',');
+
+  useEffect(() => {
+    const controller = new AbortController();
+    setIsLoading(true);
+    setIsError(false);
+
+    const params = new URLSearchParams();
+    quotableQuantities.forEach((qty, idx) => {
+      params.append(`quote[items][0][quantity][${idx}]`, String(qty));
+    });
+    params.append('quote[items][0][product_id]', String(styleId));
+    params.append('quote[color_counts][front]', String(frontColor));
+    params.append('quote[color_counts][back]', String(backColor));
+    params.append('quote[postal_code]', postalCode);
+
+    fetch(`${quotesPath}?${params.toString()}`, { signal: controller.signal })
+      .then((res) => {
+        if (!res.ok) throw new Error(`HTTP ${res.status}`);
+        return res.json() as Promise<{ mapped: Record<string, string> }>;
+      })
+      .then((data) => {
+        const mapped = Object.fromEntries(
+          quotableQuantities.map((qty) => [qty, data.mapped[String(qty)] ?? null]),
+        ) as Record<number, string | null>;
+        setPriceDetailsData(mapped);
+        setIsLoading(false);
+      })
+      .catch((err: unknown) => {
+        if (err instanceof Error && err.name === 'AbortError') return;
+        setIsError(true);
+        setIsLoading(false);
+      });
+
+    return () => controller.abort();
+    // eslint-disable-next-line react-hooks/exhaustive-deps
+  }, [styleId, frontColor, backColor, quantitiesKey, quotesPath, postalCode]);
+
+  return { isLoading, isError, priceDetailsData, quotableQuantities };
+}

+import { useEffect, useState } from 'react';
+import type { TrendingProduct } from '../ProductsCarouselWidget/types';
+import type { ApiTrendingProductsResponse } from './types';
+import { mapProduct } from '../api/mapper';
+
+const DEFAULT_BASE_URL = 'https://recommendations-service.out.customink.com';
+
+export type UseTrendingProductsByCategoryOptions = {
+  categoryId: number;
+  /** Omit to use the endpoint default (12) */
+  limit?: number;
+  /** Product IDs to exclude from results (max 100) */
+  excludeProductIds?: number[];
+  /** @default 'https://recommendations-service.out.customink.com' */
+  baseUrl?: string;
+  /** Skip fetch when false. @default true */
+  enabled?: boolean;
+  /** Design CID for logo placement in image URLs. When omitted, raw URLs are used. */
+  imageDesignCid?: string;
+};
+
+export type UseTrendingProductsByCategoryResult = {
+  products: TrendingProduct[];
+  isLoading: boolean;
+  isError: boolean;
+};
+
+export function useTrendingProductsByCategory({
+  categoryId,
+  limit,
+  excludeProductIds,
+  baseUrl = DEFAULT_BASE_URL,
+  enabled = true,
+  imageDesignCid,
+}: UseTrendingProductsByCategoryOptions): UseTrendingProductsByCategoryResult {
+  const [products, setProducts] = useState<TrendingProduct[]>([]);
+  const [isLoading, setIsLoading] = useState(enabled !== false);
+  const [isError, setIsError] = useState(false);
+
+  const excludeKey = excludeProductIds?.join(',') ?? '';
+
+  useEffect(() => {
+    if (!enabled) return;
+
+    const controller = new AbortController();
+    const url = new URL('/api/v1/trending-products/by-category', baseUrl);
+    url.searchParams.set('categoryId', String(categoryId));
+    if (limit !== undefined) {
+      url.searchParams.set('limit', String(limit));
+    }
+    if (excludeKey) {
+      url.searchParams.set('excludeProductIds', excludeKey);
+    }
+
+    setIsLoading(true);
+    setIsError(false);
+
+    fetch(url, { signal: controller.signal })
+      .then((res) => {
+        if (!res.ok) throw new Error(`HTTP ${res.status}`);
+        return res.json() as Promise<ApiTrendingProductsResponse>;
+      })
+      .then((data) => {
+        setProducts(data.trendingProducts.map((p) => mapProduct(p, { imageDesignCid })));
+        setIsLoading(false);
+      })
+      .catch((err: unknown) => {
+        if (err instanceof Error && err.name === 'AbortError') return;
+        setProducts([]);
+        setIsError(true);
+        setIsLoading(false);
+      });
+
+    return () => {
+      controller.abort();
+    };
+  }, [categoryId, limit, excludeKey, baseUrl, enabled, imageDesignCid]);
+
+  return { products, isLoading, isError };
+}
