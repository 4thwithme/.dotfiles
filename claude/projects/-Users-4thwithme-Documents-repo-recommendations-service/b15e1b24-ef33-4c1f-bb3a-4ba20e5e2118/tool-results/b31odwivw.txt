+		batches,
+		concurrency,
+		indexName,
+		progressBar,
+		onBatchComplete,
+	}: {
+		batches: { offset: number; limit: number }[];
+		concurrency: number;
+		indexName: string;
+		progressBar: ProgressBar;
+		onBatchComplete: (indexed: number, errors: number) => void;
+	}): Promise<void> {
+		await runWithConcurrencyLimit({
+			items: batches,
+			concurrency,
+			handler: async (batch) => {
+				if (this.isCancelled) {
+					return { indexed: 0, errors: 0 };
+				}

+				const styles = await this.stylesMvModel.findAll({
+					limit: batch.limit,
+					offset: batch.offset,
+				});

+				if (styles.length === 0) {
+					return { indexed: 0, errors: 0 };
+				}
+
+				return this.opensearchNeuralService.bulkIndexProducts({
+					indexName,
+					styles,
+				});
+			},
+			onItemComplete: (result) => {
+				progressBar.increment(result.indexed + result.errors);
+				onBatchComplete(result.indexed, result.errors);
+			},
+			onItemError: (error) => {
+				this.logger.error(`Batch processing error: ${(error as Error).message}`);
+			},
+		});
+	}

+	})

+		const conc = parseInt(value, 10);
+		if (isNaN(conc) || conc < 1 || conc > 20) {
+			throw new Error('Concurrency must be between 1 and 20');

+	private setupGracefulShutdown(): void {
+		const signals: NodeJS.Signals[] = ['SIGTERM', 'SIGINT', 'SIGHUP', 'SIGQUIT'];
+
+		signals.forEach((signal) => {
+			process.on(signal, () => {
+				void this.handleShutdown(signal);
+			});
+		});
+
+		process.on('uncaughtException', (error) => {
+			this.logger.error(`Uncaught exception: ${error.message}`);
+			void this.handleShutdown('SIGTERM');
+		});
+
+		process.on('unhandledRejection', (reason) => {
+			this.logger.error(`Unhandled rejection: ${String(reason)}`);
+			void this.handleShutdown('SIGTERM');
+		});
+	}
+
+	private async handleShutdown(signal: string): Promise<void> {
+		if (this.isShuttingDown) {
+			this.logger.warn('Force shutdown requested, exiting immediately');
+			process.exit(1);
+		}
+
+		this.logger.warn(`Received ${signal} signal, initiating graceful shutdown...`);
+		this.isShuttingDown = true;
+		this.isCancelled = true;
+
+		try {
+			if (this.currentOperation) {
+				this.logger.log('Waiting for current operation to complete...');
+
+				const shutdownTimeout = setTimeout(() => {
+					this.logger.error('Graceful shutdown timeout, forcing exit');
+					process.exit(1);
+				}, 30000);
+
+				try {
+					await this.currentOperation;
+				} finally {
+					clearTimeout(shutdownTimeout);
+				}
+			}
+
+			this.logger.log('Graceful shutdown completed');
+			process.exit(0);
+		} catch (error) {
+			this.logger.error(`Error during shutdown: ${(error as Error).message}`);
+			process.exit(1);
+		}
+	}

+		const brandBoosts: object[] = brand ? [{ terms: { brand: [brand] } }] : [];
+
+		const colorBoosts: object[] = detectedColor
+			? [
+					{
+						nested: {
+							path: 'colors',
+							query: {
+								wildcard: {
+									'colors.name': {
+										value: `*${detectedColor.toLowerCase()}*`,
+										case_insensitive: true,
+									},
+								},
+							},
+						},
+					},
+				]
+			: [];
+
+		const unitPriceBoosts: object[] = detectedPriceRange
+			? [
+					{
+						range: {
+							price: {
+								...(detectedPriceRange.min !== null && { gte: detectedPriceRange.min }),
+								...(detectedPriceRange.max !== null && { lte: detectedPriceRange.max }),
+							},
+						},
+					},
+				]
+			: [];
+
+		const sizeBoosts: object[] = detectedSize
+			? [
+					{
+						wildcard: {
+							sizes: { value: `*${detectedSize.toLowerCase()}*`, case_insensitive: true },
+						},
+					},
+				]
+			: [];
+
+		const hits = await this.opensearchService.search({
+			indexName: this.indexName,
+			searchPipeline: 'hybrid-search-pipeline',
+			body: {
+				size,
+				min_score: 0.4,
+				query: {
+					hybrid: {
+						queries: [
+							{
+								bool: {
+									should: [
+										{ match: { name: { query: normalizedQuery, boost: 0.1 } } },
+										{
+											match: {
+												condensed_features: { query: normalizedQuery, boost: 0.3 },
+											},
+										},
+										...priceBoosts,
+										...brandBoosts,
+										...genderBoosts,
+										...colorBoosts,
+										...sizeBoosts,
+										...unitPriceBoosts,
+									],
+									filter: baseFilters,
+								},
+							},
+							{
+								neural: {
+									product_embedding: {
+										query_text: normalizedQuery,
+										model_id: this.getModelId(),
+										k: size * 5,
+										filter: { bool: { filter: baseFilters } },
+									},
+								},
+							},
+						],
+					},
+				},
+				collapse: { field: 'product_id' },
+				_source: { excludes: ['product_embedding', 'combined_text_for_embedding'] },
+			},
+		});
+
+		const maxScore = Math.max(...hits.map((hit) => hit._score), 1);
+
+		return hits.map((hit) => {
+			const product = hit._source as IOpensearchProduct;
+			const imageUrl = detectedColor
+				? this.resolveColorImageUrl({ product, colorName: detectedColor })
+				: null;
+			const normalizedScore = hit._score / maxScore;
+
+			return {
+				...product,
+				_id: hit._id,
+				confidenceScore: Math.min(Math.round(Math.pow(normalizedScore, 2) * 100), 100),
+				...(imageUrl !== null && { style_card_image_url: imageUrl }),
+			};
+		});
+	}

+			size: params[0].size,

+			total: response.length,

+	async searchProductsByCategories({
+		categories,
+		size,
+	}: {
+		categories: Record<string, string>[];
+		size?: number;
+	}): Promise<IOpensearchProduct[]> {
+		const effectiveSize = size ?? NEURAL_SEARCH_LIMIT;
+		const perCategory = Math.ceil(effectiveSize / categories.length);
+
+		const categoryResults = await Promise.all(
+			categories.map((categoryEntry) => {
+				const categoryName = Object.keys(categoryEntry)[0] ?? '';
+				const categoryQueryText = Object.values(categoryEntry)[0] ?? '';
+
+				return this.searchProducts({
+					queryText: categoryQueryText || categoryName,
+					size: perCategory,
+					categoryFilter: categoryName,
+				});
+			}),
+		);
+
+		const interleaved: IOpensearchProduct[] = [];
+		const maxLen = Math.max(...categoryResults.map((r) => r.length));
+		for (let i = 0; i < maxLen; i++) {
+			for (const results of categoryResults) {
+				const item = results[i];
+				if (item !== undefined) {
+					interleaved.push(item);
+				}
+			}
+		}
+
+		return interleaved.slice(0, effectiveSize);
+	}

+	private resolveColorImageUrl({
+		product,
+		colorName,
+	}: {
+		product: IOpensearchProduct;
+		colorName: string;
+	}): string | null {
+		if (!product.colors?.length) return null;
+
+		const lowerColor = colorName.toLowerCase();
+		const match = product.colors.find((c) => c.name.toLowerCase() === lowerColor);
+
+		if (match?.image_url) return match.image_url;
+
+		const partialMatch = product.colors.find((c) =>
+			c.name.toLowerCase().includes(lowerColor),
+		);
+
+		return partialMatch?.image_url ?? null;
+	}

+	private normalizeCondensedFeatures({
+		condensedFeatures,
+	}: {
+		condensedFeatures: unknown;
+	}): string[] | null {
+		if (condensedFeatures === null || condensedFeatures === undefined) {
+			return null;
+		}
+
+		if (typeof condensedFeatures === 'string') {
+			return [condensedFeatures];
+		}
+
+		if (Array.isArray(condensedFeatures)) {
+			return condensedFeatures.map(String);
+		}
+
+		if (typeof condensedFeatures === 'object') {
+			return Object.values(condensedFeatures).flatMap((value: unknown) =>
+				Array.isArray(value) ? value.map(String) : [String(value)],
+			);
+		}
+
+		return [JSON.stringify(condensedFeatures)];
+	}

+	private buildCombinedTextForEmbedding({ style }: { style: IStylesMvBase }): string {
+		const normalizedCondensedFeatures = this.normalizeCondensedFeatures({
+			condensedFeatures: style.condensed_features,
+		});
+		const priceLvlMap: Record<string, string> = {
+			1: 'Budget',
+			2: 'Mid-range',
+			3: 'Premium',
+		};
+		const mappedPriceLevel: string | null =
+			style.price_level && priceLvlMap[String(style.price_level)]
+				? (priceLvlMap[String(style.price_level)] ?? null)
+				: (priceLvlMap['2'] ?? null);
+
+		const parts: string[] = [
+			...(style.name ? [this.cleanText({ text: style.name })] : []),
+			...(style.product_type ? [this.cleanText({ text: style.product_type })] : []),
+			...(mappedPriceLevel ? [`Price lvl: ${mappedPriceLevel}`] : []),
+			...(style.brand?.length
+				? [`Brand: ${this.cleanText({ text: style.brand[0] })}`]
+				: []),
+			...(normalizedCondensedFeatures?.length
+				? [this.cleanText({ text: normalizedCondensedFeatures.join(', ') })]
+				: []),
+			...(style.style_type ? [this.cleanText({ text: style.style_type })] : []),
+			...(style.primary_category_name
+				? [`Category: ${this.cleanText({ text: style.primary_category_name })}`]
+				: []),
+			...(style.manufacturer
+				? [`From: ${this.cleanText({ text: style.manufacturer })}`]
+				: []),
+			...(style.sub_category_names?.length
+				? [
+						`Subcategories: ${this.cleanText({ text: style.sub_category_names.join(', ') })}`,
+					]
+				: []),
+			...(style.decoration_method
+				? [`Decoration: ${this.cleanText({ text: style.decoration_method })}`]
+				: []),
+			...(style.features
+				? [`Features: ${this.cleanText({ text: style.features })}`]
+				: []),
+		];
+
+		const combinedText = parts.filter(Boolean).join('. ');
+
+		this.logger.debug(
+			`Combined text length for style_id ${style.style_id}: ${combinedText.length} characters`,
+		);
+
+		return combinedText;
+	}

+	private extractPrice({
+		prices,
+	}: {
+		prices: IOpensearchProduct['default_unit_prices'];
+	}): number | null {
+		if (!prices?.length) return null;
+		const first = prices[0];
+		if (!first) return null;
+		const parsed = parseFloat(first.price);
+		return isNaN(parsed) ? null : parsed;
+	}

+	private cleanText({ text }: { text: string | null | undefined }): string {

+			.replace(/\r\n|\r|\n/g, ' ')
+			.replace(/[*#]+\s?/g, '')
+			.replace(/<[^>]*>/g, '')
+			.replace(/\s{2,}/g, ' ')
+			.trim();
+	}

+			minimum_quantity: style.minimum_quantity,
+			min_qty: style.min_qty,
+			default_quote_qty: style.default_quote_qty,
+			sizes: style.sizes,
+			turn_time: style.turn_time,
+			rush_turn_time: style.rush_turn_time,
+			no_minimum: style.no_minimum ? 1 : 0,
+			price_level: style.price_level,
+			rating_score: style.rating_score !== null ? Number(style.rating_score) : null,
+			rating_count: style.rating_count,
+			default_unit_prices: style.default_unit_prices,
+			price: this.extractPrice({ prices: style.default_unit_prices }),

+			link: style.link,
+			brand: style.brand,
+			condensed_features: this.normalizeCondensedFeatures({

+				style.updated_at instanceof Date

+					: String(style.updated_at),

+					: String(style.deleted_at)

+			combined_text_for_embedding: this.buildCombinedTextForEmbedding({ style }),

+	}

+				path: `/_search/pipeline/${pipelineId}`,
+				body: {
+					description: `Hybrid search pipeline: ${normalizationTechnique} + ${combinationTechnique}`,
+					phase_results_processors: [
+						{
+							'normalization-processor': {
+								normalization: { technique: normalizationTechnique },
+								combination: { technique: combinationTechnique, parameters: { weights } },
+							},
+						},
+					],
+				},
+			});
+			this.logger.log(`Created search pipeline: ${pipelineId}`);
+		} catch (error: unknown) {
+			this.handleError({
+				context: `Failed to create search pipeline ${pipelineId}`,
+				error,
+			});
+			throw error;
+		}
+	}
+
+	@LogExecution({
+		paramFormatter: (params: [{ indexName: string; searchPipeline?: string }]) => ({
+			indexName: params[0].indexName,
+			searchPipeline: params[0].searchPipeline,
+		}),
+	})
+	async search({
+		indexName,
+		body,
+		searchPipeline,
+	}: {
+		indexName: string;
+		body: Record<string, unknown>;
+		searchPipeline?: string;
+	}): Promise<{ _id: string; _score: number; _source: Record<string, unknown> }[]> {
+		try {
+			const searchParams: Record<string, unknown> = { index: indexName, body };
+
+			if (searchPipeline) {
+				searchParams['search_pipeline'] = searchPipeline;
+			}
+
+			const response = await this.client.search(searchParams);
+
+			return (
+				response.body.hits.hits as {
+					_id: string;
+					_score: number;
+					_source: Record<string, unknown>;
+				}[]
+			).map((hit) => ({
+				_id: hit._id,
+				_score: hit._score,
+				_source: hit._source,
+			}));
+		} catch (error: unknown) {
+			this.handleError({ context: `Search failed on index ${indexName}`, error });
+			throw error;
+		}
+	}
+
+	@LogExecution()
+	async deleteIndex({ indexName }: { indexName: string }): Promise<void> {
+		try {
+			const exists = await this.client.indices.exists({ index: indexName });
+
+			if (!exists.body) {
+				this.logger.log(`Index ${indexName} does not exist`);
+				return;
+			}
+
+			await this.client.indices.delete({ index: indexName });
+			this.logger.log(`Deleted OpenSearch index: ${indexName}`);
+		} catch (error: unknown) {
+			this.handleError({ context: `Failed to delete index ${indexName}`, error });
+			throw error;
+		}
+	}

+	@LogExecution()
+	async searchModelsByGroup({
+		modelGroupId,
+	}: {
+		modelGroupId: string;
+	}): Promise<{ model_id: string; model_state: string }[]> {
+		try {
+			const response = await this.client.transport.request({
+				method: 'POST',
+				path: '/_plugins/_ml/models/_search',
+				body: {
+					query: { term: { model_group_id: modelGroupId } },
+					size: 100,
+				},
+			});
+
+			const hits = (
+				response.body as {
+					hits: { hits: { _id: string; _source: { model_state: string } }[] };
+				}
+			).hits.hits;
+
+			return hits.map((hit) => ({
+				model_id: hit._id,
+				model_state: hit._source.model_state,
+			}));
+		} catch (error: unknown) {
+			this.handleError({ context: 'Failed to search models', error });
+			throw error;
+		}
+	}
+
+	@LogExecution()
+	async searchModelGroupByName({ name }: { name: string }): Promise<string | null> {
+		try {
+			const response = await this.client.transport.request({
+				method: 'POST',
+				path: '/_plugins/_ml/model_groups/_search',
+				body: {
+					query: { term: { 'name.keyword': name } },
+					size: 1,
+				},
+			});
+
+			const hits = (response.body as { hits: { hits: { _id: string }[] } }).hits.hits;
+			return hits[0]?._id ?? null;
+		} catch (error: unknown) {
+			this.handleError({ context: `Failed to search model group: ${name}`, error });
+			throw error;
+		}
+	}

+	@LogExecution()
+	async findDeployedModelId({
+		modelGroupName,
+	}: {
+		modelGroupName: string;
+	}): Promise<string | null> {
+		const modelGroupId = await this.searchModelGroupByName({ name: modelGroupName });
+
+		if (!modelGroupId) {
+			return null;
+		}
+
+		const models = await this.searchModelsByGroup({ modelGroupId });
+		const deployed = models.find((m) => m.model_state === 'DEPLOYED');
+		return deployed?.model_id ?? null;
+	}
+
+	private isNotFoundError({ error }: { error: unknown }): boolean {
+		const message = error instanceof Error ? error.message : String(error);
+		return message.includes('resource_not_found') || message.includes('404');
+	}
+
+	private handleError({ context, error }: { context: string; error: unknown }): void {
+		const err = error instanceof Error ? error : new Error(String(error));
+		this.logger.error(`${context}: ${err.message}`, err.stack);
+	}
