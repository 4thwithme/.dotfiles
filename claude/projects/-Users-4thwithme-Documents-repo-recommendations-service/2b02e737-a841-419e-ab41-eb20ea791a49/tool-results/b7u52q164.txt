+import React from 'react';
+import { Typography } from '@customink/pigment-react';
+import styles from './MiniProductCard.module.css';
+
+export type MiniProductCardProps = {
+  name: string;
+  linkHref: string;
+  imageSrc: string;
+  imageSrcSet?: string;
+  /** @default name */
+  imageAlt?: string;
+  onClick?: (e: React.MouseEvent<HTMLAnchorElement>) => void;
+  className?: string;
+  style?: React.CSSProperties;
+};
+
+export const MiniProductCard = React.forwardRef<HTMLAnchorElement, MiniProductCardProps>(function MiniProductCard(
+  { name, linkHref, imageSrc, imageSrcSet, imageAlt, onClick, className, style },
+  ref,
+) {
+  const cardClass = [styles.card, className].filter(Boolean).join(' ');
+
+  return (
+    <a ref={ref} href={linkHref} className={cardClass} style={style} onClick={onClick} data-testid="mini-product-card">
+      <div className={styles.imageContainer}>
+        <img
+          src={imageSrc}
+          srcSet={imageSrcSet}
+          alt={imageAlt ?? name}
+          className={styles.image}
+          draggable={false}
+          loading="lazy"
+          data-testid="mini-product-card-image"
+        />
+      </div>
+      <div className={styles.content}>
+        <Typography
+          variant="heading6"
+          component="span"
+          color="primary"
+          className={styles.name}
+          title={name}
+          data-testid="mini-product-card-name"
+        >
+          {name}
+        </Typography>
+      </div>
+    </a>
+  );
+});

+import { useEffect, useState } from 'react';
+import units from '@customink/design-tokens/semantics/units';
+import type { Breakpoint } from '../../Molecules/ProductsCarousel/types';
+
+const BREAKPOINTS = {
+  mobile: parseInt(units.size.breakpoint.tablet, 10),
+  desktop: parseInt(units.size.breakpoint.desktop, 10),
+};
+
+function getBreakpoint(width: number): Breakpoint {
+  if (width < BREAKPOINTS.mobile) return 'mobile';
+  if (width < BREAKPOINTS.desktop) return 'tablet';
+  return 'desktop';
+}
+
+export function useBreakpoint(): Breakpoint {
+  const [breakpoint, setBreakpoint] = useState<Breakpoint>(() =>
+    typeof window !== 'undefined' ? getBreakpoint(window.innerWidth) : 'desktop',
+  );
+
+  useEffect(() => {
+    const mqMobile = window.matchMedia(`(max-width: ${BREAKPOINTS.mobile - 1}px)`);
+    const mqTablet = window.matchMedia(`(min-width: ${BREAKPOINTS.mobile}px) and (max-width: ${BREAKPOINTS.desktop - 1}px)`);
+    const handler = () => setBreakpoint(getBreakpoint(window.innerWidth));
+    mqMobile.addEventListener('change', handler);
+    mqTablet.addEventListener('change', handler);
+    return () => {
+      mqMobile.removeEventListener('change', handler);
+      mqTablet.removeEventListener('change', handler);
+    };
+  }, []);
+
+  return breakpoint;
+}
